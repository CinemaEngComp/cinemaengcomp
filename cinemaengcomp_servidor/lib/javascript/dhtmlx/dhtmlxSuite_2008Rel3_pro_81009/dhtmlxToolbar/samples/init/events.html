Sample: dhtmlxToolbar Event Handling dhtmlxToolbar main page 
 
<script>
 
    var webBar = new dhtmlXToolbarObject("toolbarObj");
    
    // attaching "onXLS" event handler (xml loading start)
    webBar.attachEvent("onXLS", function(id){});
    
    // attaching "onXLE" event handler (xml loading end)
    webBar.attachEvent("onXLE", function(id){});
    
    // attaching "onClick" event handler (button, buttonSelect)
    webBar.attachEvent("onClick", function(id){});
    
    // attaching "onStateChange" event handler (buttonTwoState only)
    webBar.attachEvent("onStateChange", function(id, state){});
    
    // attaching "onValueChange" event handler (slider only)
    webBar.attachEvent("onValueChange", function(id, value){});
    
    // attaching "onEnter" event handler (input only)
    webBar.attachEvent("onEnter", function(id, value){});
    
    ...
    
    webBar.loadXML("../common/dhtmlxtoolbar.xml?etc="+new Date().getTime());
</script>
