<script>
 
    var webBar = new dhtmlXToolbarObject("toolbarObj");
    ...
    // setting value for a slider
    webBar.setValue(itemId, value);
    // getting value of a slider
    var value = webBar.getValue(itemId);
    // setting minimal value for a slider
    webBar.setMinValue(itemId, value, label);
    // getting minimal value of a slider
    var data = webBar.getMinValue(itemId); // data = Array(value, label)
    // setting maximal value for a slider
    webBar.setValue(itemId, value, label);
    // getting maximal value of a slider
    var data = webBar.getValue(itemId); // data = Array(value, label)
</script>
