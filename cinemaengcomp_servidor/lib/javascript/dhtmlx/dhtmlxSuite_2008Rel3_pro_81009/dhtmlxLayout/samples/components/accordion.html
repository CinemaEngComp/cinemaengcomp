Sample: dhtmlxLayout Attach dhtmlxAccordion to Layout dhtmlxLayout main page 
 
<script>
 
    // initialization of the layout
    var dhxLayout = new dhtmlXLayoutObject("parentId", "3L");
    // attaching an accordion
    var dhxAccord = dhxLayout.cells("a").attachAccordion();
    dhxAccord.addItem("a1", "Accordion Panel A");
    dhxAccord.addItem("a2", "Accordion Panel B");
    dhxAccord.addItem("a3", "Accordion Panel C");
    dhxAccord.openItem("a1");
</script>
