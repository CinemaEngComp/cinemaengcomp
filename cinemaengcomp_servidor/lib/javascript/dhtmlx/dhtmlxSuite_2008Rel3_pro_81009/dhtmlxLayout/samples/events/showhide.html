Sample: dhtmlxLayout Expand/Collapse Event Handling dhtmlxLayout main page 
 
 
<script>
 
    // initialization of the layout
    var dhxLayout = new dhtmlXLayoutObject("parentId", "5I");
    ...
    // attaching onExpand event handler
    dhxLayout.attachEvent("onExpand", function(ietmId){
        alert("Item "+itemId+" was expanded");
    });
    // attaching onCollapse event handler
    dhxLayout.attachEvent("onCollapse", function(ietmId){
        alert("Item "+itemId+" was collapsed");
    });
</script>
