Sample: dhtmlxGrid Configuring combo boxes on the fly dhtmlxGrid main page

The comboboxes have different values for even and uneven rows. (the Shipping column) 


 
<div id="gridbox" width="100%" height="250px" style="background-color:white;overflow:hidden"></div>
<script>
 
    function myf(stage,id,index){
        var ind=mygrid.getRowIndex(id);
        if ((index==5)&&(stage==0))//start edit Shipping column
            {
            var combo=mygrid.getCombo(5);
            if (ind%2==1){                      //for even rows
                combo.save();                   //save initial state
                combo.remove(1);
                combo.remove(2);
                combo.remove(3);
                }
                else{
                combo.save();                   //save initial state
                combo.remove(4);
                combo.remove(5);
                combo.remove(6);
                }
            }
 
        if ((index==5)&&(stage==2))             //for finishing edit
           mygrid.getCombo(5).restore();        //restore combo state
        return true;
    }
    mygrid = new dhtmlXGridObject('gridbox');
    mygrid.setImagePath("../../codebase/imgs/");
    mygrid.attachEvent("onEditCell",myf);
    mygrid.loadXML("gridH.xml");
</script>
