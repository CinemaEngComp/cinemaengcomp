Sample: dhtmlxGrid Context menu for specific column dhtmlxGrid main page

In example below context menu is available only for first column.

 
 
<div id="gridbox" width="100%" height="250px" style="background-color:white;overflow:hidden"></div>
<script>
 
    function onButtonClick(menuitemId,type){
        var data = mygrid.contextID.split("_"); //rowId_colInd
        mygrid.setRowTextStyle(data[0],"color:"+menuitemId.split("_")[1]);
        return true
    }
    function my_pre_func(rowId,celInd,grid){
        if (celInd==1) return false;
        return true;
    }
    menu = new dhtmlXMenuObject();
    menu.setImagePath("../../../dhtmlxMenu/codebase/imgs/standard/");
    menu.renderAsContextMenu();
    menu.setOpenMode("web");
    menu.attachEvent("onClick",onButtonClick);
    menu.loadXML("_context.xml");
 
    mygrid = new dhtmlXGridObject('gridbox');
    ...
    mygrid.enableContextMenu(menu);
    
        mygrid.attachEvent("onBeforeContextMenu",my_pre_func);
        
    mygrid.init();
    mygrid.loadXML("grid_links.xml");
</script>
