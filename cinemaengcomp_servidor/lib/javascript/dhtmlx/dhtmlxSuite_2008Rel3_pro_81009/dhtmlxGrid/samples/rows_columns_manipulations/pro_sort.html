Sample: dhtmlxGrid Custom sorting routine dhtmlxGrid main page

Use your own comparator function to sort rows in grid in some special way.
Second column is sorted by number of symbols in text. 


 
<div id="gridbox" width="100%" height="250px" style="background-color:white;overflow:hidden"></div>
<script>
 
    //sort by last name
    function sort_custom(a,b,order){
        var n=a.length;
        var m=b.length;
    if(order=="asc")
        return n>m?1:-1;
    else
        return n<m?1:-1;
    }
 
    mygrid = new dhtmlXGridObject('gridbox');
    ...
    //set sort_custom function as sorting routine for second column
    mygrid.setColSorting("int,sort_custom,str,int,str,str,str,date")
    //Another way how to set comparator. Set sort_custom function as sorting routine for third column        
    mygrid.setCustomSorting(sort_custom,2);
    ...
    mygrid.init();
    mygrid.loadXML("grid.xml");
</script>
