<script>
 
    mygrid = new dhtmlXGridObject('gridbox');
    mygrid.setColTypes("edn,edn,clist,edn,acheck,calendar");
    
    mygrid.setNumberFormat("@000.00Zbh",1);
    mygrid.setNumberFormat("%script%,000",0);
    mygrid.setDateFormat("d/m [y]");
    
    mygrid.init();
    mygrid.loadXML("grid_format.xml");
